            {
                   h=h%12;
                   let hour = (h*360)/12 + (m*360)/(12*60);
                   let min = (m*360)/60;
                   let angle = Math.abs(hour-min);
                   
                   if(angle>180)
                   return 360-angle;
                   else
                   return angle;
            };

Perfect Number Check

            var Perfect_Check = (N) => 
            {
              let sum = 0;
                  for(i=0;i<=N;i++){
                    if(N%i===0){
                    sum = sum+i;
                  }
                }
                if (sum===N){
                  return "YES";
                }
                else{
                  return "NO";
                }
            };

Reverse the Number

            var Reverse_Number = (N) => 
            {
              let result = 0;
              while(N>0){
                rightmostNo = N%10;
                result = result*10+rightmostNo;
                N=Math.floor(N/10);
              }
              return result;
            
            };
